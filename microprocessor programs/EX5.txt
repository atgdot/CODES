.MODEL SMALL
.STACK 64
.DATA
	NUM DW 3
	RES DW 2 DUP(0)
.CODE
	MOV AX,@DATA
	MOV DS,AX
	MOV AX,NUM
	CALL FACT
	MOV AH,4CH
	INT 21H
FACT PROC NEAR
	CMP AX,01H
	JA CONT
	MOV RES,01
	RET
CONT:	PUSH AX
	DEC AX
	CALL FACT
	POP AX
	MUL AX
	MUL RES
	MOV RES,AX
	MOV RES+2,DX
	RET
FACT 	ENDP
	END
