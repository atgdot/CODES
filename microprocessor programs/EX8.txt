.MODEL SMALL
.DATA
	CR EQU 13
	LF EQU 10
	PASSWORD DB 'MIRCROPROCESSOR'
	PROMPT DB 'ENTER PASSWORD:','$'
	ENTRY DB 0FFH,?,40 DUP(?)
	MSG DB CR,LF,'PASSWORD CORRECT','$'
	NOMSG DB CR,LF,'WRONG PASSWORD!','$'

.CODE
	MOV,@DATA
	MOV DS,AX
	MOV ES,AX
	MOV CX,0EH

	LEA DX,PROMPT
	MOV AH,09H
	INT 21H
	
	MOV DX,OFFSET ENTRY
	MOV AH,0AH
	INT 21H
	
	LEA SI,PASSWORD
	LEA DI,ENTRY
	ADD DI,02
	MOV CX,0EH
	
	REPE CMPSB
	JE SUC
	LEA DX,NOMSG
	MOV AH,04H
	INT 21H
	
	JMP OVER
	SUC:LEA DX,MSG
	MOV AH,09H
	INT 21H

OVER: MOV AH,4CH
	INT 21H
	END
