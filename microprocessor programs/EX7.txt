.MODEL SMALL
.DATA
	ASC_STR1 DB '0354'
	ASC_ST2 DB '9375'
	SSIZE EQU ASC_ST2-ASC_STR1-1
	ADD_RES DB SSIZE+2 DUP(0)
	SUB_RES DB SSIZE+2 DUP(0)
.CODE
	MOV AX,@DATA
	MOV DS,AX
	MOV SI,OFFSET ASC_STR1+SSIZE
	MOV DI,OFFSET ASC_ST2+SSIZE
		
	PUSH SI
	PUSH DI
	MOV CX,SSIZE+1
	MOV BX,OFFSET ADD_RES
	MOV BP,OFFET SUB_RES
	
	XOR AH,AH
BACK1:	MOV AL,[SI]
	ADC AL,[DI]
	AAA
	MOV [BX],AL
	DEC SI
	DEC DI
	INC BX
	LOOP BACK1
	
	MOV AH,00
	RCL AH,1
	MOV [BX],AL
	
	POP DI
	POP SI
	MOV CX,SSIZE+1
	XOR AH,AH
BACK2:	MOV AL,[SI]
	SBB AL,[DI]
	AAS
	MOV DS:[BP],AL
	DEC SI
	DEC DI
	INC BP
	LOOP BACK2

	MOV AH,4CH
	INT 21H
	END
