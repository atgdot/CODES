.MODEL SMALL
.STACK200
.DATA
	BCD_INPUT DB 45H,67H
	HEX_VALUE DW 0
	FACTORS DB 0AH,64H,0E8H,03H
	NB_SEP DB 10H
.CODE
	MOV AX,@DATA
	MOV DS,AX
	MOV SI,OFFSET BCD_INPUT
	MOV DI,OFFSET HEX_VALUE
	CALL BCD_HEX
	MOV AH,4CH
	INT 21H
	BCD_HEX PROC NEAR
	MOV AX,WORD PTR[SI]
	OR AH,00
	JZ BYTE_CONV
	MOV BL,AL
	MOV AH,0
	DIV NB_SEP
	MOV CH,AH
	MOV AH,0
	MUL WORD PTR FACTORS+2
	MOV WORD PTR HEX_VALUE,AX
	MOV AL,CH
	MUL FACTORS+1
	ADD HEX_VALUE,AX
	MOV AL,BL
	MOV AH,0
BYTE_CONV:DIV NB_SEP
	MOV DL,AH
	MOV DH,0
	MUL FACTORS
	ADD AX,DX
	ADD HEX_VALUE,AX
	RET
  BCD_HEX ENDP
	END
