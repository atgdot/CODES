.MODEL SMALL
.DATA
	NUM DB 7DH
	RES DB DUP(?)
	N_IS DB 00H
	N_OS DB 08H
.CODE
	MOV AX,@DATA
	MOV DS,AX
	LEA SI,NUM
	LEA DI,RES
	MOV AX,[SI]
	MOV CX,08
BACK:	ROL AL,1
	JNC LOC1
	INC N_IS
LOC1:	LOOP BACK
	MOV AL,N_IS
	MOV BYTE PTR [DI],AL
	SUB N_OS,AL
	MOV AL,N_OS
	MOV BYTE PTR 1[DI],AL
	MOV AH,4CH
	INT 21H
	END
