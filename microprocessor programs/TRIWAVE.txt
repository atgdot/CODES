	OUTPUT 2500AD
	ORD 2000H
	MOV AX,0000H
	MOV CS,AX
	MOV ES,AX
	MOV DX,0FFE6H
	MOV AL,80H
	OUT DX,AL

START:    MOV CX,0FFH
	MOV AL,00H
   UP: 	INC AL
	MOV DX,0FFE0H
	OUT DX,AL
	MOV DX,0FFE2H
	OUT DX,AL
	LOOP UP
	MOV CX,0FFH
	MOV AX,CX
 DOWN: 	 DEC AL
	MOV DX,0FFE0H
	OUT DX,AL
	MOV DX,0FFE2H
	OUT DX,AL
	LOOP DOWN
	JMP SHORT START
	END
