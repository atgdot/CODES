.MODEL SMALL
.DATA
	N1 DW 1239H
	N2 DW 0ADC2H
	RES DW 4 DUP(0)
	DIFF DW 4 DUP(0)
.CODE
START:    MOV AX,@DATA
	  MOV DS,AX
	MOV AX,N1
	MOV BX,N2
	MOV DX,0000H

	ADD AX,BX
	JNC LAB1
	INC DX
LAB1:	  MOV RES,AX
	MOV RES+2,DX
	
	MOV AX,N1
	MOV BX,N2
	MOV DX,0000H
	SUB AX,BX
	JNC LAB2
	MOV DX,00FFH
LAB2:	  MOV DIFF,DX
	MOV DIFF+2,AX

	MOV AH,4CH
	INT 21H
	END
